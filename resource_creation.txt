#!/bin/bash

set -e

# Set environment variables
REGION="centralindia"
RGP="day11-demo-rg"
CLUSTER_NAME="day11-demo-cluster"
ACR_NAME="ahad306acr"
SQLSERVER="ahad306sqlserver"
DB="mhcdb"

# Function to handle errors
handle_error() {
    echo "Error: $1"
    exit 1
}

# Function to check if the resource exists
resource_exists() {
    az resource show --ids "$1" &> /dev/null
}

echo "Deleting Azure resources..."

# Delete AKS
AKS_ID=$(az aks show \
    --resource-group "$RGP" \
    --name "$CLUSTER_NAME" \
    --query id \
    --output tsv 2>/dev/null)

if [ -n "$AKS_ID" ] && resource_exists "$AKS_ID"; then
    az aks delete \
        --resource-group "$RGP" \
        --name "$CLUSTER_NAME" \
        --yes \
        --no-wait || handle_error "Failed to delete AKS."
else
    echo "AKS not found. Skipping."
fi

# Delete ACR
ACR_ID=$(az acr show \
    --name "$ACR_NAME" \
    --resource-group "$RGP" \
    --query id \
    --output tsv 2>/dev/null)

if [ -n "$ACR_ID" ] && resource_exists "$ACR_ID"; then
    az acr delete \
        --name "$ACR_NAME" \
        --resource-group "$RGP" \
        --yes || handle_error "Failed to delete ACR."
else
    echo "ACR not found. Skipping."
fi

# Delete SQL Database
DB_ID=$(az sql db show \
    --resource-group "$RGP" \
    --server "$SQLSERVER" \
    --name "$DB" \
    --query id \
    --output tsv 2>/dev/null)

if [ -n "$DB_ID" ] && resource_exists "$DB_ID"; then
    az sql db delete \
        --resource-group "$RGP" \
        --server "$SQLSERVER" \
        --name "$DB" \
        --yes || handle_error "Failed to delete SQL Database."
else
    echo "SQL Database not found. Skipping."
fi

# Delete SQL Server
SQL_ID=$(az sql server show \
    --resource-group "$RGP" \
    --name "$SQLSERVER" \
    --query id \
    --output tsv 2>/dev/null)

if [ -n "$SQL_ID" ] && resource_exists "$SQL_ID"; then
    az sql server delete \
        --resource-group "$RGP" \
        --name "$SQLSERVER" \
        --yes || handle_error "Failed to delete SQL Server."
else
    echo "SQL Server not found. Skipping."
fi

# Delete Resource Group
RG_ID=$(az group show \
    --name "$RGP" \
    --query id \
    --output tsv 2>/dev/null)

if [ -n "$RG_ID" ] && resource_exists "$RG_ID"; then
    az group delete \
        --name "$RGP" \
        --yes \
        --no-wait || handle_error "Failed to delete Resource Group."
else
    echo "Resource Group not found. Skipping."
fi

echo "Delete operation initiated successfully."
